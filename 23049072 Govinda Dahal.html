def summary_statistics(df):
    numeric_df = df.select_dtypes(include=['float64', 'int64'])

    summary = pd.DataFrame({
        'Sum': numeric_df.sum(),
        'Mean': numeric_df.mean(),
        'Standard Deviation': numeric_df.std(),
        'Skewness': numeric_df.skew(),
        'Kurtosis': numeric_df.kurt()
    })
    print("\nSummary Statistics:\n")
    print(summary)

import pandas as pd
from scipy.stats import f_oneway

data = {
    'complaint_type': ['A', 'B', 'A', 'C', 'B', 'C', 'A', 'B', 'C', 'A'],
    'response_time': [10, 12, 15, 8, 11, 9, 14, 13, 7, 16]
}
df = pd.DataFrame(data)


groups = [df[df['complaint_type'] == typ]['response_time'] for typ in df['complaint_type'].unique()]


f_stat, p_value = f_oneway(*groups)


print("Test 1: One-way ANOVA")
print(f"Null Hypothesis (H): Average response time is the same across all complaint types.")
print(f"Alternative Hypothesis (H): At least one complaint type has a different average response time.")
print(f"F-statistic: {f_stat:.4f}, p-value: {p_value:.4f}")

if p_value < 0.05:
    print("Result: Reject H (significant difference exists).")
else:
    print("Result: Fail to reject H (no significant difference).")

from scipy.stats import chi2_contingency


data = {
    'complaint_type': ['A', 'B', 'A', 'C', 'B', 'C', 'A', 'B', 'C', 'A'],
    'location': ['X', 'Y', 'X', 'Y', 'X', 'Y', 'X', 'Y', 'X', 'Y']
}
df = pd.DataFrame(data)

contingency_table = pd.crosstab(df['complaint_type'], df['location'])

chi2_stat, p_value, dof, expected = chi2_contingency(contingency_table)


print("\nTest 2: Chi-Square Test of Independence")
print(f"Null Hypothesis (H): Complaint type and location are independent.")
print(f"Alternative Hypothesis (H): Complaint type and location are related.")
print(f"Chi2-statistic: {chi2_stat:.4f}, p-value: {p_value:.4f}")

if p_value < 0.05:
    print("Result: Reject H (variables are related).")
else:
    print("Result: Fail to reject H (no significant relationship).")
